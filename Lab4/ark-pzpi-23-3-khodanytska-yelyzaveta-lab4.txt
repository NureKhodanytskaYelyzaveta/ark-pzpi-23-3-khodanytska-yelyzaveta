 378     # –ó–∞–ø—É—Å–∫–∞—î —Ç–∞–π–º–µ—Ä     
 379     def start_pickup_timer(self):
 380         timeout = 60
 381         
 382         def countdown():
 383             if not self.current_session['unlock_time']:
 384                 return
 385                 
 386             remaining = timeout - int(time.time() - self.current_session['unlock_time'])
 387             
 388             if remaining <= 0:
 389                 self.root.after(0, self.confirm_pickup_dialog)
 390             else:
 391                 timer_color = self.colors['danger'] if remaining < 10 else self.colors['primary']
 392                 self.timer_label.config(
 393                     text=f"‚è≥ –õ–∏—à–∏–ª–æ—Å—è {remaining} —Å–µ–∫ –Ω–∞ —Ç–µ, –∞–±–∏ –∑–∞–±—Ä–∞—Ç–∏ –∫–Ω–∏–≥—É",
 394                     fg=timer_color
 395                 )
 396                 self.root.after(1000, countdown)
 397         
 398         countdown()

 424     # –ê—Å–∏–Ω—Ö—Ä–æ–Ω–Ω–æ –ø—ñ–¥—Ç–≤–µ—Ä–¥–∂—É—î –≤–∏–¥–∞—á—É –∫–Ω–∏–≥–∏
 425     def _confirm_pickup_async(self):
 426         locker_id = self.current_session['locker_id']
 427         try:
 428             response = requests.post(
 429                 f"{self.server_url}/iot/lockers/confirm_pickup",
 430                 json={
 431                     "user_id": self.current_session['user_id'],
 432                     "book_id": self.current_session['book_id']
 433                 },
 434                 timeout=5
 435             )
 436             
 437             if response.status_code == 200:
 438                 data = response.json()
 439                 loan_id = data.get('loan_id')
 440                 due_date = data.get('due_date')
 441                 
 442                 self.root.after(0, lambda: self.update_status(
 443                     f"üéâ –£—Å–ø—ñ—à–Ω–æ!\n–ü–æ–∑–∏–∫–∞ #{loan_id}\n–¢–µ—Ä–º—ñ–Ω: {due_date[:10]}\n\n–ü—Ä–∏—î–º–Ω–æ–≥–æ —á–∏—Ç–∞–Ω–Ω—è!",

 451                     "–£—Å–ø—ñ—Ö!",
 452                     f"–ö–Ω–∏–≥—É –≤–∏–¥–∞–Ω–æ!\n–ü–æ–∑–∏–∫–∞ #{loan_id}\n"
 453                     f"–ü–æ–≤–µ—Ä–Ω—É—Ç–∏ –¥–æ: {due_date[:10]}"
 454                 ))
